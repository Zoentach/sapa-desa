CREATE TABLE user (
    id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
    full_name TEXT NOT NULL,
    email TEXT NOT NULL,
    kode_desa TEXT NOT NULL,
    kode_kecamatan TEXT NOT NULL,
    token TEXT NOT NULL,
    mac_address TEXT
);

CREATE TABLE perangkat_desa (
    id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
    nama TEXT,
    nipd TEXT,
    kode_kecamatan TEXT,
    kode_desa TEXT,
    kode_jabatan TEXT,
    mulai TEXT,
    berakhir TEXT,
    nik TEXT,
    tempat_lahir TEXT,
    tanggal_lahir TEXT,
    sk_id TEXT,
    pendidikan TEXT,
    jenis_kelamin TEXT,
    agama TEXT,
    no_telp TEXT,
    status INTEGER
);
